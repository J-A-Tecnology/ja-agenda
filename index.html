// Garante que o script do Google rode apenas após a página carregar 100% window.addEventListener('load', function () { // 1. Inicia o funcionamento do Relógio Digital startClock(); // 2. Inicializa a API do Google Auth // ⚠️ SUBSTITUA A STRING ABAIXO PELO SEU CLIENT ID REAL DO GOOGLE CLOUD google.accounts.id.initialize({ client_id: "SUAS_CREDENCIAIS_AQUI.apps.googleusercontent.com", callback: handleCredentialResponse }); // 3. Renderiza (desenha) o botão dentro do elemento #buttonDiv google.accounts.id.renderButton( document.getElementById("buttonDiv"), { theme: "outline", size: "large", text: "signin_with", shape: "rectangular" } ); // Exibe o prompt flutuante One Tap do Google no canto da tela (opcional) google.accounts.id.prompt(); }); // Função executada automaticamente assim que o usuário faz o login function handleCredentialResponse(response) { // Decodifica o token criptografado vindo do Google para extrair Nome e Foto const dataBase64 = response.credential.split('.')[1]; const jsonString = window.atob(dataBase64); const usuario = JSON.parse(jsonString); console.log("Usuário Autenticado com sucesso:", usuario.name); // Remove o botão de login e exibe a foto de perfil com uma saudação amigável const authContainer = document.getElementById("google-auth-container"); authContainer.innerHTML = `
Olá, ${usuario.given_name}!
`; } // Faz o relógio digital funcionar em tempo real function startClock() { setInterval(() => { const now = new Date(); const timeString = now.toTimeString().split(' ')[0]; const clockElement = document.getElementById('live-clock'); if (clockElement) { clockElement.textContent = timeString; } }, 1000); } // Ação do botão verde "Adicionar na Agenda" function salvarTarefa() { const taskName = document.getElementById('task-name').value; if (!taskName) { alert("Por favor, digite o que você precisa fazer!"); return; } alert(`Tarefa "${taskName}" guardada com sucesso!`); document.getElementById('task-name').value = ""; // Limpa a barra de texto }
